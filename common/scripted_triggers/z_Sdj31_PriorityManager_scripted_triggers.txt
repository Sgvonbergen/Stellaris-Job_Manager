SdjTriggerJobPriorityN_job_criminal = {
    planet = {
        OR = {
            NOT = { check_variable = { which = SdjVarJobPriority_job_criminal value > -0.5 } }
            
            AND = {
                prev.species = { has_species_flag = SdjFlagForbidden_job_criminal_@prevprev.owner }
                NOT = { has_planet_flag = SdjFlagDeregulationOfJob@prev.owner }
            }
        }
    }
}
